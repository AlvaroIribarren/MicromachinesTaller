title Threads

class Thread{
    -thread thread
    +void start()
    +virtual void join();
    +virtual void run();
    +virtual ~Thread();
}

class ReceiverThread {
    +virtual void run();
    +virtual ~ReceiverThread();
}

class SenderThread {
    +virtual void run();
    +virtual ~SenderThread();
}

class AcceptingThread {
    +virtual void run();
    +virtual ~AcceptingThread();
}



class Room {
    +virtual void run();
    +virtual ~Room();
}

class Client {
    +virtual void run();
    +virtual ~Client();
}
