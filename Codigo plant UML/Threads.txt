title Threads

class Thread{
    -thread thread
    +void start()
    +virtual void join();
    +virtual void run();
    +virtual ~Thread();
}

class ReceiverThread {
    +virtual void run();
    +virtual ~ReceiverThread();
}

class SenderThread {
    +virtual void run();
    +virtual ~SenderThread();
}

class AcceptingThread {
    +virtual void run();
    +virtual ~AcceptingThread();
}



class GameThread {
    +virtual void run();
    +virtual ~GameThread();
}

class Collector {
    +virtual void run();
    +virtual ~Collector();
}

class LobbyListener {
    +virtual void run();
    +virtual ~LobbyListener();
}
